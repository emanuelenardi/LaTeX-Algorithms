\prototype{\Bool \subSetSumRec{\Array{Int} \(A\), \Int \(i\), \Int \(r\)}}{

	\BlankLine
	\uIf{\(r \Equal 0\)}{
		\Return \True\;
		\BlankLine
	}
	\uElseIf{\(i \Equal 0\)}{
		\Return \False\;
		\BlankLine
	}
	\uElseIf{\(A[i] > r\)}{
		\Return \(\subSetSumRec(A, i-1, r)\)\;
		\BlankLine
	}
	\Else{
		\Return \(\subSetSumRec(A, i-1, r)\) \Or \(\subSetSumRec(A, i-1, r-A[i])\)\;
	}

	\BlankLine
	\Return \(res\)\;
}
